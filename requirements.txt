# Runtime dependencies. Note what is NOT here: torch, torchvision, basicsr,
# facexlib, gfpgan, realesrgan, modelscope, datasets, timm.
#
# The models run as ONNX, exported ahead of time (see requirements-export.txt and
# tools/export_onnx.py). Dropping torch took the whole pin cascade with it —
# numpy<2, timm<1.0, datasets<3, torchvision==0.16.2 and the Python 3.10 ceiling
# are all gone. That is what makes this installable alongside another service
# instead of fighting it for versions.

# --- Telegram ---
aiogram==3.29.1
python-dotenv==1.2.2

# --- Inference ---
onnxruntime==1.27.0

# --- Images ---
# Headless: no GUI libraries, so a container needs no libGL.
# It also ships the YuNet face detector (cv2.FaceDetectorYN), which is why the bot
# needs neither a separate detection model nor torch-based facexlib.
opencv-python-headless==5.0.0.93
numpy==2.4.6
Pillow==12.3.0

# HEIC/HEIF — what iPhones shoot by default. Neither OpenCV nor Pillow reads it
# unaided, and without this the bot rejects a large share of what people send.
pillow-heif==1.4.0
